🏆 GPA รายนักศึกษา
คำนวณจาก AVG + CASE WHEN grade → GPA points
SELECT s.student_id, s.name, s.major,
       COUNT(e.id) AS courses_taken,
       ROUND(AVG(CASE e.grade
         WHEN 'A'  THEN 4.0
         WHEN 'B+' THEN 3.5
         WHEN 'B'  THEN 3.0
         ...
       END), 2) AS gpa
FROM students s
LEFT JOIN enrollments e ON s.student_id = e.student_id
GROUP BY s.student_id, s.name, s.major
ORDER BY gpa DESC NULLS LAST

📚 สถิติรายวิชา
จำนวนนักศึกษาที่ลงทะเบียนและจำนวน Grade A
SELECT c.course_id, c.course_name,
       COUNT(e.student_id) AS enrolled_count,
       SUM(CASE WHEN e.grade = 'A' THEN 1 ELSE 0 END) AS grade_a_count
FROM courses c
LEFT JOIN enrollments e ON c.course_id = e.course_id
GROUP BY c.course_id, c.course_name
ORDER BY enrolled_count DESC
